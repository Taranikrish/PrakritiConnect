Corrected Firestore Security Rules:

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // Allow users to read/write their own data in users collection
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow users to read/write their own data in organizers collection  
    match /organizers/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to read messages
    match /messages/{docId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to read events, but only organizers can create/update/delete their own events
    match /events/{eventId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null 
                   && request.auth.uid == request.resource.data.organizerId;
      allow update: if request.auth != null 
                   && (
                     // Organizers can update their own events
                     request.auth.uid == resource.data.organizerId ||
                     // Volunteers can update participant fields only
                     (request.auth.uid != resource.data.organizerId && 
                      request.writeFields.hasOnly(['participants', 'participantIds']))
                   );
      allow delete: if request.auth != null 
                   && request.auth.uid == resource.data.organizerId;
    }
    
    // Allow all authenticated users to read organizers collection (for the Organizations page)
    match /organizers/{orgId} {
      allow read: if request.auth != null; // Allow read access to all authenticated users
      allow write: if request.auth != null && request.auth.uid == orgId; // Only allow users to write their own data
    }
  }
}
